        a1 = AND inputs 2;
        a2 = AND inputs 2;
        SW1 = SWITCH;
        SW2 = SWITCH;

        clock - a1.I2;

        a2 - o1.I2;
    }
    SIGNALS{
        SW1 = 1;

    }

NETWORK{
    DEVICES{
        clock = CLOCK halfperiod 5;
        D1 = DTYPE;
        D2 = DTYPE;
        nand = NAND inputs 1;
        SW = SWITCH;
    }
    CONNECTIONS{
        clock - D1.CLK;
        SW - D1.DATA;
        clock - nand.I1;
        nand - D2.CLK;
        D1.Q - D2.DATA; 
    }
    SIGNALS{
        SW = 0;

    }
    MONITOR{
        clock;
        D1.Q;
        D2.Q;
    }
}
